**Tech Stack:** Rust 2021、tokio、serde/serde_json、async_trait、reqwest、inventory、现有 Tool/Command 系统。

```rust
pub struct McpServerConfig {
    #[serde(default)]
    pub command: Option<String>,
    #[serde(default)]
    pub args: Vec<String>,
    #[serde(default)]
    pub env: HashMap<String, String>,
    #[serde(default)]

    pub headers: HashMap<String, String>,
    #[serde(default)]
    pub transport: Option<McpTransportKind>,
}

#[derive(Debug, Clone, Copy, Serialize, Deserialize, PartialEq, Eq)]
#[serde(rename_all = "snake_case")]
pub enum McpTransportKind {
    Stdio,
    Sse,
    StreamableHttp,
}

impl McpServerConfig {
    pub fn transport_kind(&self) -> Result<McpTransportKind, String> {
        if let Some(kind) = self.transport {
            return Ok(kind);
        }
        if self.command.as_deref().is_some_and(|s| !s.trim().is_empty()) {
            return Ok(McpTransportKind::Stdio);
        }
        if self.url.as_deref().is_some_and(|s| !s.trim().is_empty()) {
            return Ok(McpTransportKind::StreamableHttp);
        }
        Err("MCP server config must define either command or url".to_string())
    }
}
```

- [ ] **Step 4: 添加安全函数**

```rust
pub fn validate_remote_url(url: &str) -> Result<(), String> {
    let parsed = url::Url::parse(url).map_err(|e| format!("invalid MCP url: {e}"))?;
    match parsed.scheme() {
        "https" => Ok(()),
        "http" => {
            let host = parsed.host_str().unwrap_or_default();
            if host == "localhost" || host == "127.0.0.1" || host == "::1" {
                Ok(())
            } else {
                Err("remote MCP url must use https unless it points to localhost".to_string())
            }
        }
        other => Err(format!("unsupported MCP url scheme: {other}")),
    }
}

pub fn redact_headers(headers: &HashMap<String, String>) -> HashMap<String, String> {
    headers
        .iter()
        .map(|(key, value)| {
            let lower = key.to_ascii_lowercase();
            let sensitive = lower == "authorization"
                || lower == "cookie"
                || lower == "x-api-key"
                || lower == "proxy-authorization";
            if sensitive {
                (key.clone(), "<redacted>".to_string())
            } else {
                (key.clone(), value.clone())
            }
        })
        .collect()
}
```

```rust
pub const DEFAULT_MAX_TOOL_RESPONSE_BYTES: usize = 1_048_576;

pub fn limit_tool_response(output: &str, max_bytes: usize) -> String {
    if output.len() <= max_bytes {
        return output.to_string();
    }
    let mut end = max_bytes.min(output.len());
    while !output.is_char_boundary(end) {
        end -= 1;
    }
    format!(
        "{}\n\n[Output truncated: original {} bytes, limit {} bytes]",
        &output[..end],
        output.len(),
        max_bytes
    )
}
```

```rust
#[derive(Debug, Clone, PartialEq)]
pub struct ToolListDiff {
    pub added: Vec<McpToolDef>,
    pub removed: Vec<String>,
    pub changed: Vec<McpToolDef>,
}

pub fn diff_tools(old: &[McpToolDef], new: &[McpToolDef]) -> ToolListDiff {
    let old_map: HashMap<&str, &McpToolDef> = old.iter().map(|t| (t.name.as_str(), t)).collect();
    let new_map: HashMap<&str, &McpToolDef> = new.iter().map(|t| (t.name.as_str(), t)).collect();

    let added = new
        .iter()
        .filter(|tool| !old_map.contains_key(tool.name.as_str()))
        .cloned()
        .collect();
    let removed = old
        .iter()
        .filter(|tool| !new_map.contains_key(tool.name.as_str()))
        .map(|tool| tool.name.clone())
        .collect();
    let changed = new
        .iter()
        .filter(|tool| {
            old_map
                .get(tool.name.as_str())
                .is_some_and(|old_tool| old_tool.description != tool.description || old_tool.input_schema != tool.input_schema)
        })
        .cloned()
        .collect();

    ToolListDiff { added, removed, changed }
}
```

```rust
use aemeath_core::mcp_manager::{McpConnectionManager, McpManagerConfig};
use std::collections::HashMap;
use std::sync::Arc;

pub fn parse_mcp_servers_config(
    config: &serde_json::Value,
) -> Result<HashMap<String, McpServerConfig>, String> {
    let Some(servers) = config.get("mcpServers").and_then(|v| v.as_object()) else {
        return Ok(HashMap::new());
    };

    let mut parsed = HashMap::new();
    for (name, server_config) in servers {
        let cfg: McpServerConfig = serde_json::from_value(server_config.clone())
            .map_err(|e| format!("invalid MCP server config '{}': {e}", name))?;
        parsed.insert(name.clone(), cfg);
    }
    Ok(parsed)
}

pub fn merge_mcp_servers(
    base: &mut HashMap<String, McpServerConfig>,
    overlay: HashMap<String, McpServerConfig>,
) {
    for (name, config) in overlay {
        base.insert(name, config);
    }
}
```

```rust
pub async fn load_mcp_manager(cwd: &PathBuf) -> Arc<McpConnectionManager> {
    let config_paths = [
        dirs::home_dir()
            .map(|h| h.join(".aemeath").join("mcp.json"))
            .unwrap_or_default(),
        cwd.join(".mcp.json"),
    ];

    let mut servers = HashMap::new();
    for config_path in &config_paths {
        if !config_path.exists() {
            continue;
        }
        let content = match tokio::fs::read_to_string(config_path).await {
            Ok(content) => content,
            Err(e) => {
                log::warn!("failed to read MCP config {}: {e}", config_path.display());
                continue;
            }
        };
        let value: serde_json::Value = match serde_json::from_str(&content) {
            Ok(value) => value,
            Err(e) => {
                log::warn!("invalid MCP config {}: {e}", config_path.display());
                continue;
            }
        };
        match parse_mcp_servers_config(&value) {
            Ok(parsed) => merge_mcp_servers(&mut servers, parsed),
            Err(e) => log::warn!("invalid MCP config {}: {e}", config_path.display()),
        }
    }

    let manager = Arc::new(McpConnectionManager::with_servers(servers));
    if let Err(e) = manager.initialize().await {
        log::warn!("failed to initialize MCP manager: {e}");
    }
    manager
}
```

```rust
pub fn execute_mcp_command(args: &str, ctx: &mut CommandContext) -> CommandResult {
    let parts: Vec<&str> = args.split_whitespace().collect();
    match parts.as_slice() {
        [] => CommandResult::Success(format_mcp_servers_from_config(ctx)),
        ["tools"] => CommandResult::Success(format_mcp_tools_from_config(ctx, None)),
        ["tools", server] => CommandResult::Success(format_mcp_tools_from_config(ctx, Some(server))),
        ["restart", server] => CommandResult::Success(format!("MCP restart requested for server '{}'. Runtime bridge will reconnect it.", server)),
        ["add", name, target, rest @ ..] => CommandResult::Success(format_mcp_add_preview(name, target, rest)),
        ["remove", name] => CommandResult::Success(format!("MCP remove requested for server '{}'. Runtime bridge will disconnect and remove it.", name)),
        [unknown, ..] => CommandResult::Error(format!("Unknown MCP command: {}", unknown)),
    }
}

fn format_mcp_servers_from_config(_ctx: &CommandContext) -> String {
    "MCP servers:\n  No MCP servers configured in command context".to_string()

    match server {
        Some(name) => format!("MCP tools for '{}':\n  Runtime MCP manager status is not attached to command context yet", name),
        None => "MCP tools:\n  Runtime MCP manager status is not attached to command context yet".to_string(),
    }
}

fn format_mcp_add_preview(name: &str, target: &str, rest: &[&str]) -> String {
```

```rust
pub async fn mark_reconnecting(&self, name: &str) -> Result<(), String> {
    let mut connections = self.connections.lock().await;
    let Some(connection) = connections.get_mut(name) else {
        return Err(format!("Server '{}' not found", name));
    };
    connection.state = ConnectionState::Reconnecting;
    connection.error = None;
    Ok(())
}

pub async fn refresh_tool_snapshot(&self, name: &str, tools: Vec<McpToolDef>) {
    let mut discovered = self.discovered_tools.lock().await;
    discovered.insert(name.to_string(), tools.clone());
    drop(discovered);

    let mut connections = self.connections.lock().await;
    if let Some(connection) = connections.get_mut(name) {
        connection.tools = tools;
    }
}
```

```rust
pub async fn health_check_once(&self) {
    let connected = self.get_connected_servers().await;
    for connection in connected {
        let Some(client) = connection.client else {
            continue;
        };
        let ping_result = client.lock().await.ping().await;
        if let Err(e) = ping_result {
```
